9.3.2 Tuned XGB

# Set seed for reproducibility
set.seed(12345)

# Set up control parameters
ctrl <- trainControl(
  method = 'cv',
  number = 10,
  verboseIter = TRUE
  )

# Define the parameter grid
grid <- expand.grid(
  nrounds = c(100, 250, 500),
  max_depth = c(3, 6, 9),
  eta = c(0.01, 0.1, 0.3),
  gamma = 0,
  colsample_bytree = 1,
  min_child_weight=1,
  subsample=1
  )
# Tune the model
xgb_tuned <- train(
  x = as.matrix(train_xgb_x),
  y = as.factor(train_xgb_y),
  method = 'xgbTree',
  trControl = ctrl,
  tuneGrid = grid
  )

# Retrieve best model
model_xgb_tuned <- xgb_tuned$finalModel
# Display summary
print(xgb_tuned)
